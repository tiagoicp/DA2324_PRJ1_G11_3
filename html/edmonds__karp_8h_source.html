12/* This file contains an adapted Edmonds Karp algorithm for our graph.
13 * It also allows for reversing the order of adj and incoming vectors.
14 * Time Complexity = O(V*E^2) */
15
16bool findAugmentingPath(WaterSupply &g, Vertex<string> *source, Vertex<string> *target, bool inverse_order) {
17 for (Vertex<string> *v: g.getNodeSet()) {
18 v->setVisited(false);
19 }
20 source->setVisited(true);
21 std::queue<Vertex<string> *> q;
22 q.push(source);
23 while (!q.empty()) {
24 Vertex<string> *v = q.front();
25 q.pop();
26 if (inverse_order) v->reverseAdj();
27 for (Edge<string> *e: v->getAdj()) {
28 Vertex<string> *w = e->getDest();
29 if (!w->isVisited() && e->getWeight() > e->getFlow()) {
30 q.push(w);
31 w->setPath(e);
32 w->setVisited(true);
33 }
34 }
35 if (inverse_order) v->reverseIncoming();
36 for (Edge<string> *e: v->getIncoming()) {
37 Vertex<string> *w = e->getOrig();
38 if (!w->isVisited() && e->getFlow() > 0) {
39 q.push(w);
40 w->setPath(e);
41 w->setVisited(true);
42 }
43 }
44 }
45 return (target->isVisited());
46}

48void edmondsKarp(WaterSupply &g, Vertex<string> *s, Vertex<string> *t, bool inverse_order) {
49 while (findAugmentingPath(g, s, t, inverse_order)) {
50 double minFlow = INF;
51 for (Vertex<string> *v = t; v != s;) {
52 Edge<string> *e = v->getPath();
53 if (e->getDest() == v) {
54 minFlow = std::min(e->getWeight() - e->getFlow(), minFlow);
55 v = e->getOrig();
56 } else {
57 minFlow = std::min(e->getFlow(), minFlow);
58 v = e->getDest();
59 }
60 }
61 for (Vertex<string> *v = t; v != s;) {
62 Edge<string> *e = v->getPath();
63 if (e->getDest() == v) {
64 e->setFlow(e->getFlow() + minFlow);
65 v = e->getOrig();
66 } else {
67 e->setFlow(e->getFlow() - minFlow);
68 v = e->getDest();
69 }
70 }
71 }
72}
